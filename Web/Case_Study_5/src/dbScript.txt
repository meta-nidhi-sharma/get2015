DROP DATABASE vehicle_management_system;

CREATE DATABASE vehicle_management_system;
USE vehicle_management_system;
DELETE FROM `vehicle_management_system`.`vehicle` WHERE `vehicle_id`='123345';
CREATE TABLE vehicle
(
vehicle_id int,
make VARCHAR(10),
model VARCHAR(50),
engineIn_CC INTEGER,
fuel_capacity double,
milage double,
price double,
road_tax double,
created_by VARCHAR(20),
created_time TIMESTAMP NOT NULL Default NOW(),
PRIMARY KEY(vehicle_id)
);

CREATE TABLE bike
(
vehicle_id INTEGER,
self_start VARCHAR(5) CHECK (selfStart IN ('True','False')),
helmet_price INTEGER,
FOREIGN KEY(vehicle_id) REFERENCES vehicle(vehicle_id),
PRIMARY KEY(vehicle_id)
);

CREATE TABLE car
(
vehicle_id INTEGER, 
ac VARCHAR(5) CHECK (AC IN ('True','False')),
power_stearing VARCHAR(5) CHECK (power_stearing IN ('True','False')),
accessory_kit VARCHAR(5) CHECK (accessory_kit IN ('True','False')),
FOREIGN KEY(vehicle_id) REFERENCES vehicle(vehicle_id),
PRIMARY KEY(vehicle_id)
);

CREATE TABLE user
(
userName VARCHAR(20),
password VARCHAR(20),
PRIMARY KEY(userName)
);
